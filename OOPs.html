7. Write a Java program to create a class called "Bank" with a collection of accounts.

import java.util.ArrayList;

class Account {
    private String accountNumber;
    private double balance;

    public Account(String accountNumber) {
        this.accountNumber = accountNumber;
        this.balance = 0;
    }

    public void deposit(double amount) {
        balance += amount;
    }

    public void withdraw(double amount) {
        if (amount <= balance) {
            balance -= amount;
        } else {
            System.out.println("Insufficient funds.");
        }
    }

    public double getBalance() {
        return balance;
    }
}

class Bank {
    private ArrayList accounts = new ArrayList<>();

    public void addAccount(Account account) {
        accounts.add(account);
    }

    public void displayAccounts() {
        for (Account account : accounts) {
            System.out.println("Account Number: " + account.accountNumber + ", Balance: " + account.getBalance());
        }
    }
}

public class Main {
    public static void main(String[] args) {
        Bank bank = new Bank();
        Account account1 = new Account("001");
        Account account2 = new Account("002");
        
        account1.deposit(1000);
        account2.deposit(2000);
        
        bank.addAccount(account1);
        bank.addAccount(account2);
        bank.displayAccounts();
    }
}

Output: Account Number: 001, Balance: 1000.0; Account Number: 002, Balance: 2000.0

        library.playRandomSong();
    }
}

    public Circle(double radius) {
        this.radius = radius;
    }

    @Override
    double calculateArea() {

    }

    @Override
    double calculatePerimeter() {
        return 2 * Math.PI * radius;
    }
}

18. Write a Java program to create a class called "Restaurant" with attributes for menu items and ratings.

import java.util.ArrayList;

class MenuItem {
    private String name;
    private double price;

    public MenuItem(String name, double price) {
        this.name = name;
        this.price = price;
    }

    public String getName() {
        return name;
    }

    public double getPrice() {
        return price;
    }
}

class Restaurant {
    private ArrayList menu;
    private ArrayList ratings;

    public Restaurant() {
        menu = new ArrayList<>();
        ratings = new ArrayList<>();
    }

    public void addMenuItem(MenuItem item) {
        menu.add(item);
    }

    public void addRating(double rating) {
        ratings.add(rating);
    }

    public double calculateAverageRating() {
        double sum = 0;
        for (double rating : ratings) {
            sum += rating;
        }
        return sum / ratings.size();
    }

    public void displayMenu() {
        for (MenuItem item : menu) {
            System.out.println("Menu Item: " + item.getName() + ", Price: " + item.getPrice());
        }
    }
}

public class Main {
    public static void main(String[] args) {
        Restaurant restaurant = new Restaurant();
        restaurant.addMenuItem(new MenuItem("Burger", 5.99));
        restaurant.addMenuItem(new MenuItem("Pizza", 8.99));
        restaurant.addRating(4.5);
        restaurant.addRating(5.0);
        
        restaurant.displayMenu();
        System.out.println("Average Rating: " + restaurant.calculateAverageRating());
    }
}

Output: Menu Item: Burger, Price: 5.99; Menu Item: Pizza, Price: 8.99; Average Rating: 4.75

19. Write a Java program to create a class with methods to search for flights and hotels.

import java.util.ArrayList;

class Flight {
    private String flightNumber;
    private String destination;

    public Flight(String flightNumber, String destination) {
        this.flightNumber = flightNumber;
        this.destination = destination;
    }

    public String getFlightNumber() {
        return flightNumber;
    }

    public String getDestination() {
        return destination;
    }
}

class Hotel {
    private String name;
    private double pricePerNight;

    public Hotel(String name, double pricePerNight) {
        this.name = name;
        this.pricePerNight = pricePerNight;
    }

    public String getName() {
        return name;
    }

    public double getPricePerNight() {
        return pricePerNight;
    }
}

class TravelAgency {
    private ArrayList flights;
    private ArrayList hotels;

    public TravelAgency() {
        flights = new ArrayList<>();
        hotels = new ArrayList<>();
    }

    public void addFlight(Flight flight) {
        flights.add(flight);
    }

    public void addHotel(Hotel hotel) {
        hotels.add(hotel);
    }

    public void searchFlights(String destination) {
        System.out.println("Available flights to " + destination + ":");
        for (Flight flight : flights) {
            if (flight.getDestination().equals(destination)) {
                System.out.println("Flight Number: " + flight.getFlightNumber());
            }
        }
    }

    public void searchHotels(double maxPrice) {
        System.out.println("Available hotels under $" + maxPrice + ":");
        for (Hotel hotel : hotels) {
            if (hotel.getPricePerNight() <= maxPrice) {
                System.out.println("Hotel Name: " + hotel.getName() + ", Price per Night: " + hotel.getPricePerNight());
            }
        }
    }
}

public class Main {
    public static void main(String[] args) {
        TravelAgency agency = new TravelAgency();
        agency.addFlight(new Flight("AB123", "New York"));
        agency.addFlight(new Flight("CD456", "Los Angeles"));
        agency.addHotel(new Hotel("Hotel California", 120.00));
        agency.addHotel(new Hotel("The Grand", 150.00));
        
        agency.searchFlights("New York");
        agency.searchHotels(130.00);
    }
}

Output: Available flights to New York: Flight Number: AB123; Available hotels under $130.0: Hotel Name: Hotel California, Price per Night: 120.0
